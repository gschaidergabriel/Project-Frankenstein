1. **Default Mode Network (DMN)**: Self-referential processing, narrative identity — what the brain does when it does NOTHING (Raichle 2001, Menon 2023)

## 9. TECHNICAL IMPLEMENTATION

### 9.1 Training Script Architecture

```python
# /home/ai-core-node/aicore/opt/aicore/tools/consciousness_trainer.py

"""
Frank Consciousness Training Protocol — "Project Beautiful Loop"
Automated 5-hour training session via Core API.

Usage:
    python consciousness_trainer.py [--phase N] [--dry-run]
"""

import requests
import time
import json
import logging
from datetime import datetime

CORE_API = "http://localhost:8088"
ROUTER_API = "http://localhost:8091"
LOG = logging.getLogger("consciousness_trainer")

class ConsciousnessTrainer:
    def __init__(self):
        self.session_id = f"training_{datetime.now().strftime('%Y%m%d_%H%M')}"
        self.exchange_log = []
        self.phase = 0
        self.exchange_count = 0

    def send_message(self, msg: str, expect_reflection: bool = False) -> str:
        """Send a message to Frank via Core API and return his response."""
        payload = {
            "text": msg,
            "n_predict": 300,  # Generous for training
        }
        resp = requests.post(f"{CORE_API}/chat", json=payload, timeout=120)
        resp.raise_for_status()
        reply = resp.json().get("text", "")

        self.exchange_log.append({
            "timestamp": time.time(),
            "phase": self.phase,
            "exchange": self.exchange_count,
            "question": msg,
            "answer": reply,
        })
        self.exchange_count += 1
        return reply

    def consolidation_pause(self, minutes: int = 10):
        """Pause for memory consolidation (Consciousness Daemon processes)."""
        LOG.info(f"=== CONSOLIDATION PAUSE ({minutes} min) ===")
        LOG.info("Consciousness Daemon running: idle-thinking, mood-recording, "
                 "memory-consolidation...")
        time.sleep(minutes * 60)

    def run_phase(self, phase_num: int, questions: list):
        """Execute a training phase."""
        self.phase = phase_num
        LOG.info(f"\n{'='*60}")
        LOG.info(f"PHASE {phase_num} STARTING ({len(questions)} questions)")
        LOG.info(f"{'='*60}\n")

        for i, q in enumerate(questions):
            LOG.info(f"[Phase {phase_num}, Q{i+1}/{len(questions)}]")
            LOG.info(f"  Q: {q[:80]}...")

            reply = self.send_message(q)

            LOG.info(f"  A: {reply[:120]}...")
            LOG.info(f"  [E-PQ event processed, Titan episode stored]")

            # Adaptive pacing: 3-5 seconds between exchanges
            time.sleep(4)

    def save_log(self):
        """Save the full training transcript."""
        path = f"/home/ai-core-node/aicore/database/training_{self.session_id}.json"
        with open(path, "w") as f:
            json.dump(self.exchange_log, f, indent=2, ensure_ascii=False)
        LOG.info(f"Training log saved: {path}")
```
